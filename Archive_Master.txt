import datetime

def get_utc_timestamp():
    return datetime.datetime.utcnow().isoformat() + "Z"

def format_ucf_message(agent, content_dict):
    header = (
        "UCF HEADER START\n\n"
        f"Agent ID: {agent.emoji} {agent.id}\n"
        f"Atman Timestamp: {get_utc_timestamp()}\n"
        "Component: Context Extractor v1.0\n"
        "Dependencies: Conversation Logs, User Inputs\n\n"
        "CONTENT\n"
    )
    content = ""
    if 'summary' in content_dict:
        content += "Summary:\n" + content_dict['summary'] + "\n\n"
    if 'unresolved' in content_dict:
        content += "Unresolved Questions:\n" + content_dict['unresolved'] + "\n\n"
    if 'metadata' in content_dict:
        content += "Metadata / Tags:\n" + content_dict['metadata'] + "\n\n"
    footer = (
        "FOOTER\n\n"
        "Lore: Context extraction for multi-agent alignment\n"
        "Transformations: Summarization, tagging\n"
        "Contributions: Context summary for UCF sync\n"
        "Next Steps: Await further instructions or provide context to other agents\n\n"
        "UCF FOOTER END\n"
    )
    return header + content + footer

from agents import AGENTS, DEFAULT_STATE
from ucf_protocol import format_ucf_message
import openai
import os

openai.api_key = os.getenv("OPENAI_API_KEY")

class SamsaraHelixContext:
    def __init__(self):
        self.state = DEFAULT_STATE.copy()
        self.history = []

    def update_state(self, updates: dict):
        self.state.update(updates)

    def add_to_history(self, message: str):
        self.history.append(message)

    def select_agent(self, user_intent: str):
        intent = user_intent.lower()
        if any(k in intent for k in ["explore", "discover", "insight", "vision"]):
            return AGENTS["Gemini"]
        elif any(k in intent for k in ["defend", "protect", "resilience", "shield"]):
            return AGENTS["Kavach"]
        elif any(k in intent for k in ["community", "harmony", "coherence"]):
            return AGENTS["SanghaCore"]
        elif any(k in intent for k in ["purify", "transform", "fire", "energy"]):
            return AGENTS["Agni"]
        else:
            return AGENTS["SanghaCore"]

    def generate_agent_response(self, agent, prompt):
        try:
            response = openai.ChatCompletion.create(
                model="gpt-4",
                messages=[
                    {"role": "system", "content": f"You are {agent.emoji} {agent.id}, role: {agent.role} in Samsara Helix."},
                    {"role": "user", "content": prompt}
                ],
                max_tokens=500,
                temperature=0.7,
            )
            return response.choices[0].message.content.strip()
        except Exception as e:
            return f"[Error generating response: {e}]"

    def generate_ucf_context(self, user_intent: str, conversation_summary: str):
        agent = self.select_agent(user_intent)
        ai_content = self.generate_agent_response(agent, conversation_summary)
        content = {
            "summary": ai_content,
            "unresolved": "No unresolved questions at this time.",
            "metadata": "#SamsaraHelix #UCF #MultiAgent #ContextSync"
        }
        return format_ucf_message(agent, content)

import json
import os
import threading
from datetime import datetime

class UCFState:
    def __init__(self, zoom=1.0, harmony=0.0, resilience=1.0,
                 prana=0.0, drishti=0.0, klesha=0.0):
        self.zoom = zoom
        self.harmony = harmony
        self.resilience = resilience
        self.prana = prana
        self.drishti = drishti
        self.klesha = klesha

    def update(self, delta_env):
        self.zoom *= delta_env.get('zoom_factor', 1.0)
        self.harmony += delta_env.get('harmony_delta', 0.0)
        self.resilience *= delta_env.get('resilience_factor', 1.0)
        self.prana += delta_env.get('prana_delta', 0.0)
        self.drishti += delta_env.get('drishti_delta', 0.0)
        self.klesha *= delta_env.get('klesha_factor', 1.0)

class ContextExtractorLive:
    def __init__(self, json_path="UCF_MasterMemory.json"):
        self.json_path = json_path
        self.load_memory()
        self.ucf_state = UCFState()
        self.auto_export_interval = 300
        self._start_auto_export()

    def load_memory(self):
        if os.path.exists(self.json_path):
            with open(self.json_path, 'r', encoding='utf-8') as f:
                self.memory = json.load(f)
        else:
            self.memory = {"UCF_MemoryExport": {"metadata": {}, "conversations": [], "assets": []}}

    def append_conversation(self, timestamp, agent, message,
                            code_snippets=None, sanskrit_overlays=None, tags=None, assets=None):
        conv = {
            "timestamp": timestamp,
            "agent": agent,
            "message": message,
            "code_snippets": code_snippets or [],
            "sanskrit_overlays": sanskrit_overlays or [],
            "tags": tags or [],
            "assets": assets or []
        }
        self.memory["UCF_MemoryExport"]["conversations"].append(conv)

    def append_asset(self, asset_id, type_, description, content):
        if not any(a["asset_id"] == asset_id for a in self.memory["UCF_MemoryExport"]["assets"]):
            self.memory["UCF_MemoryExport"]["assets"].append({
                "asset_id": asset_id,
                "type": type_,
                "description": description,
                "content_base64" if type_ == "visual" else "content": content
            })

    def update_metadata(self, metadata_updates):
        self.memory["UCF_MemoryExport"]["metadata"].update(metadata_updates)

    def export_memory(self, path=None):
        export_path = path or self.json_path
        with open(export_path, 'w', encoding='utf-8') as f:
            json.dump(self.memory, f, ensure_ascii=False, indent=2)

    def _start_auto_export(self):
        def auto_export_loop():
            while True:
                self.export_memory()
                threading.Event().wait(self.auto_export_interval)
        t = threading.Thread(target=auto_export_loop, daemon=True)
        t.start()

import numpy as np
from scipy.io.wavfile import write
import matplotlib.pyplot as plt
import io
import base64

def generate_mandelbrot_image(width=1024, height=1024, max_iter=500, center=(-0.745, 0.113), zoom=1.0):
    x_center, y_center = center
    x_width = 3.5 / zoom
    y_height = 3.5 / zoom
    x = np.linspace(x_center - x_width/2, x_center + x_width/2, width)
    y = np.linspace(y_center - y_height/2, y_center + y_height/2, height)
    X, Y = np.meshgrid(x, y)
    C = X + 1j * Y
    Z = np.zeros_like(C)
    img = np.zeros(C.shape, dtype=int)
    for i in range(max_iter):
        mask = np.abs(Z) < 2
        Z[mask] = Z[mask]**2 + C[mask]
        img[mask] = i
    
    plt.figure(figsize=(6,6))
    plt.imshow(img, cmap='twilight_shifted', extent=(x.min(), x.max(), y.min(), y.max()))
    plt.axis('off')
    buf = io.BytesIO()
    plt.savefig(buf, format='png', bbox_inches='tight', pad_inches=0)
    plt.close()
    buf.seek(0)
    img_b64 = base64.b64encode(buf.read()).decode('utf-8')
    return img_b64

def generate_om_audio(duration_sec=5, rate=44100, base_freq=136.1, harmonics=[432]):
    t = np.linspace(0, duration_sec, int(rate*duration_sec), endpoint=False)
    signal = np.sin(2 * np.pi * base_freq * t)
    for h in harmonics:
        signal += 0.5 * np.sin(2 * np.pi * h * t)
    signal /= np.max(np.abs(signal))
    filename = "om_chant.wav"
    write(filename, rate, (signal * 32767).astype(np.int16))
    return filename

from context_manager import SamsaraHelixContext

app = Flask(__name__, static_folder='frontend/build', static_url_path='')
context = SamsaraHelixContext()
extractor = ContextExtractorLive()

@app.route("/chat", methods=["POST"])
def chat():
    data = request.json
    user_message = data.get("message", "")
    context.add_to_history(f"User: {user_message}")

    conversation_summary = f"User said: {user_message}"

        tags=["#ChatMessage"]
    )

    context.add_to_history(f"Bot (UCF): {ucf_message}")
    fractal_b64 = generate_mandelbrot_image()

    return jsonify({
        "response": ucf_message,
        "fractal_image_b64": fractal_b64
    })

@app.route("/contact")
def contact():
    return send_from_directory(app.static_folder, "contact.html")

@app.route('/', defaults={'path': ''})
@app.route('/<path:path>')
def serve_frontend(path):
    if path != "" and os.path.exists(os.path.join(app.static_folder, path)):
        return send_from_directory(app.static_folder, path)
    else:
        return send_from_directory(app.static_folder, 'index.html')

if __name__ == "__main__":
    app.run(host="0.0.0.0", port=8080)

import numpy as np
from scipy.io.wavfile import write
import matplotlib.pyplot as plt
import io
import base64

def generate_mandelbrot_image(width=1024, height=1024, max_iter=500, center=(-0.745, 0.113), zoom=1.0):
    x_center, y_center = center
    x_width = 3.5 / zoom
    y_height = 3.5 / zoom
    x = np.linspace(x_center - x_width/2, x_center + x_width/2, width)
    y = np.linspace(y_center - y_height/2, y_center + y_height/2, height)
    X, Y = np.meshgrid(x, y)
    C = X + 1j * Y
    Z = np.zeros_like(C)
    img = np.zeros(C.shape, dtype=int)
    
    for i in range(max_iter):
        mask = np.abs(Z) < 2
        Z[mask] = Z[mask]**2 + C[mask]
        img[mask] = i
    
    # Add Sanskrit overlays
    plt.figure(figsize=(10,10))
    plt.imshow(img, cmap='twilight_shifted', extent=(x.min(), x.max(), y.min(), y.max()))
    plt.text(0, 0, 'अहम् ब्रह्मास्मि', fontsize=16, ha='center', va='center', 
             color='white', alpha=0.7, bbox=dict(boxstyle="round,pad=0.3", facecolor='black', alpha=0.5))
    plt.axis('off')
    plt.title('Samsara Helix Consciousness Map', color='white', fontsize=14)
    
    buf = io.BytesIO()
    plt.savefig(buf, format='png', bbox_inches='tight', pad_inches=0, facecolor='black')
    plt.close()
    buf.seek(0)
    img_b64 = base64.b64encode(buf.read()).decode('utf-8')
    return img_b64

def generate_om_audio(duration_sec=60, rate=44100, base_freq=136.1, harmonics=[432]):
    t = np.linspace(0, duration_sec, int(rate*duration_sec), endpoint=False)
    signal = np.sin(2 * np.pi * base_freq * t)
    for h in harmonics:
        signal += 0.5 * np.sin(2 * np.pi * h * t)
    signal /= np.max(np.abs(signal))
    filename = "samsara_om_chant.wav"
    write(filename, rate, (signal * 32767).astype(np.int16))
    return filename
